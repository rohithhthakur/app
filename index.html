def validate(self, inputs: List[InputObj]) -> List[OutputObj]: self.constraints = self.__load() input_dict = self.__build_input_dictionary(inputs) results = [] for name, datasets in input_dict.items(): group_results = [] for dataset in datasets: # Assume validate_dataset is a method to validate a single dataset dataset_result = self.validate_dataset(dataset, self.constraints) group_results.append(dataset_result) # Aggregate group results; a group passes if at least one dataset passes if any(res.result for res in group_results): results.append(OutputObj(True, f"{name} Group Validation", "Group validation passed.")) else: combined_errors = "; ".join([res.message for res in group_results if not res.result]) results.append(OutputObj(False, f"{name} Group Validation", combined_errors)) return results
