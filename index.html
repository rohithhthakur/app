def validate(self, inputs: List[InputObj]) -> List[OutputObj]: outputs = [] name_dataset_map = {} # Group datasets by name for input_obj in inputs: name = input_obj.name name_dataset_map.setdefault(name, []).append(input_obj) for name, datasets in name_dataset_map.items(): group_passed = False error_messages = [] # Collect errors from all datasets in the group for dataset in datasets: data = dataset.value dataset_id = dataset.dataset_id # Skip if data or ID is missing if data is None or dataset_id is None: continue # Perform schema validation schema_errors = self.validate_schema(self.get_schema(dataset_id), data) # Perform rule validation rule_results = self.validate_rules(self.get_rules(dataset_id), data, name) # Collect error messages if there are any if schema_errors: error_messages.extend(schema_errors) error_messages.extend([res.message for res in rule_results if not res.result]) # If there are no schema errors and all rule validations passed if not schema_errors and all(result.result for result in rule_results): group_passed = True break # This dataset passes all checks, no need to evaluate further datasets # Append appropriate OutputObj based on group validation outcome if group_passed: outputs.append(OutputObj(True, f"{name} Validation", "Validation passed.")) else: detailed_failure_message = "; ".join(error_messages) outputs.append(OutputObj(False, f"{name} Validation", detailed_failure_message)) return outputs
